// VAL STRING

const schema = {
  image: 'string'
}

const schema = {
  image: {
    type: 'string'
  }
};

const schema = {
  image: {
    type: 'string',
    required: true
  }
};




// ARRAY

const schema = {
  images: 'array'
}

const schema = {
  images: {
    type: 'array'
  }
};

const schema = {
  images: {
    type: 'array',
    required: true
  }
};




// VAL OBJECT

const schema = {
  image: 'object' // WRONG WAY
}

const schema = {
  image: {
    type: 'object', // GOOD WAY
    properties: {
      srcset: 'string'
    }
  }
};

const schema = {
  image: {
    type: 'object',
    properties: {
      srcset: {
        type: 'string',
        required: true
      }
    },
    required: true
  }
};


// MULTI TYPES

const schema = {
  image: ['string', 'boolean']
};
// just is accepted: 'string', 'boolean', 'number' or 'array', but never 'object', because a property type 'object' can't be declare with 'simple way' nor in a 'multiple types'.

const schema = {
  image: ['string', 'object'] // WRONG WAY!!!
};
